  (for*/first ([i input]
               [j input]
               #:when (= (+ i j) 2020))
    (* i j)))

    # 
    def fix_expense_report(data):
    for i in data:
        for j in data:
            if i + j == 2020:
                return i * j

data = [int(x) for x in input_data.split()]
